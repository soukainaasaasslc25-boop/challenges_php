<?php

        // Fonction de validation
        function validerDonnees($depart, $arrivee, $typeEnvoi, $poids) {
            $erreurs = [];
            
            // Validation ville de départ
            $villes_valides = ['Casablanca', 'Rabat', 'Tanger'];
            if (empty($depart) || !in_array($depart, $villes_valides)) {
                $erreurs[] = "Ville de départ invalide";
            }
            
            // Validation ville d'arrivée
            if (empty($arrivee) || !in_array($arrivee, $villes_valides)) {
                $erreurs[] = "Ville d'arrivée invalide";
            }
            
            // Validation type d'envoi
            $types_valides = ['Standard', 'Express'];
            if (empty($typeEnvoi) || !in_array($typeEnvoi, $types_valides)) {
                $erreurs[] = "Type d'envoi invalide";
            }
            
            // Validation poids
            if (empty($poids) || !is_numeric($poids) || $poids <= 0) {
                $erreurs[] = "Le poids doit être un nombre strictement positif";
            }
            
            return $erreurs;
        }

        // Fonction de calcul du tarif
        function calculerTarif($depart, $arrivee, $typeEnvoi, $poids) {
            $tarif = 20; // Tarif de base

            // Vérification si les villes sont différentes
            if ($depart !== $arrivee) {
                $tarif *= 1.5; // Augmentation de 50%
            }

            // Application de la surtaxe pour le poids
            if ($poids > 5) {
                $tarif += ($poids - 5) * 5; // +5 DH par kg au-dessus de 5kg
            }

            // Application de la majoration pour l'option Express
            if ($typeEnvoi === 'Express') {
                $tarif *= 1.2; // Augmentation de 20%
            }

            return round($tarif, 2); // Arrondi à 2 décimales
        }

        
        if ($_SERVER["REQUEST_METHOD"] == "POST") {
           
            $depart = $_POST['depart'] ?? '';
            $arrivee = $_POST['arrivee'] ?? '';
            $typeEnvoi = $_POST['typeEnvoi'] ?? '';
            $poids = $_POST['poids'] ?? '';

       
            $erreurs = validerDonnees($depart, $arrivee, $typeEnvoi, $poids);

            if (empty($erreurs)) {
                // Conversion du poids en float
                $poids = floatval($poids);
                
                // Calcul du tarif
                $tarif = calculerTarif($depart, $arrivee, $typeEnvoi, $poids);
                
                // Affichage du résultat
                echo '<div class="result">';
                echo '<h2>Devis de livraison</h2>';
                echo '<p><strong>Ville de départ :</strong> ' . htmlspecialchars($depart) . '</p>';
                echo '<p><strong>Ville d\'arrivée :</strong> ' . htmlspecialchars($arrivee) . '</p>';
                echo '<p><strong>Poids :</strong> ' . htmlspecialchars($poids) . ' kg</p>';
                echo '<p><strong>Livraison :</strong> ' . htmlspecialchars($typeEnvoi) . '</p>';
                echo '<p class="total">Total à payer : ' . htmlspecialchars($tarif) . ' DH</p>';
                echo '</div>';
            } else {
                // Affichage des erreurs
                echo '<div class="error">';
                echo '<strong>Erreurs de validation :</strong><br>';
                foreach ($erreurs as $erreur) {
                    echo '• ' . htmlspecialchars($erreur) . '<br>';
                }
                echo '</div>';
            }
        } else {
            // Si quelqu'un accède directement à la page sans soumettre le formulaire
            echo '<div class="error">';
            echo 'Aucune donnée reçue. Veuillez remplir le formulaire.';
            echo '</div>';
        }
        ?>
        
        <a href="index.html" class="back-button">← Nouveau calcul</a>
    </div>
</body>
</html>
